CREATE OR REPLACE PACKAGE PKG_SIMPLE_SUM AS
  FUNCTION SIMPLE_SUM RETURN NUMBER;
END PKG_SIMPLE_SUM;

/

CREATE OR REPLACE PACKAGE BODY PKG_SIMPLE_SUM AS
  FUNCTION SIMPLE_SUM RETURN NUMBER IS
    a number;
    b number;
BEGIN
    a := 7;
    b := 77;
    DBMS_OUTPUT.PUT_LINE('Sum of the number is: ' || (a+b));
  END;
END PKG_SIMPLE_SUM;
